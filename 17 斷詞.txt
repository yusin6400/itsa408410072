#include <stdio.h>
#include <string.h>
#include <ctype.h>

char *getword(char *text,char *word)
{
    char *ptr=text;
    char *qtr=word;
    word[0]='\0';
    while(isspace(*ptr)) ptr++;
    while(!isspace(*ptr) && *ptr!='\0')
    {
        if(*ptr >= 'A' && *ptr <= 'Z') *ptr=*ptr+32; //轉小寫
        *qtr++=*ptr++;
    }
    *qtr='\0';
    if(word[0]=='\0') return NULL;
    return ptr;
}
